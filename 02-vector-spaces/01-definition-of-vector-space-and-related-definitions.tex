% TODO: Przy pomocy algebry uniwersalnej}

\begin{definition}
	Niech $V, W$ będą przestrzeniami liniowymi nad ciałem $\mathbb{K}$
		oraz niech $f: V \to W$ będzie przekształceniem liniowym.
	Jądrem przekształcenia $f$ nazywamy zbiór
	\begin{equation}
		\ker f = \{v \in V : f(v) = 0\}.
	\end{equation}
\end{definition}
